\section{Ejercicios}

DNI: 77149477W


\begin{ejer}
	Sea (X,Y) un vector aleatorio continuo con la función de densidad conjunta que se muestra a continuación:
	
	$$ f(x,y)= \frac{4}{289}, \ \ y < x < 1-y, \ \ -8 < y < 0.5$$
	
	Obtener la función de densidad de $y$ condicionada a un valor $x_0$ , así como la función de densidad de $x$ condicionada a un valor $y_0$. A través de esas funciones de densidad condicionadas, calcular $P(Y > -3.38|X=3.81)$ y $P(X<3.81|Y=-3.38)$.
\end{ejer}

\begin{sol}.
	
	$$\includegraphics{foto22}$$
	
	 Calculo primero la función de densidad marginal de X:
	 
	 \begin{equation}
	 f_1(x) = \left\lbrace
	 \begin{array}{ll}
	 \displaystyle\int_{-8}^{x} \frac{4}{289} \ dy = \frac{4}{289} \displaystyle\int_{-8}^{x} 1 \ dy = \frac{4}{289}(x+8) = \frac{4}{289}x+\frac{32}{289}, \ \ -8 < x < 0.5 \\ 
	 \displaystyle\int_{-8}^{1-x} \frac{4}{289} \ dy = \frac{4}{289} \displaystyle\int_{-8}^{1-x} 1 \ dy = \frac{4}{289}(9-x) = \frac{36}{289}-\frac{4}{289}x, \ \ 0.5 \leq x < 9 \\ 
	 \end{array}
	 \right. 
	 \end{equation}
	
	Por otra parte, calculo la función de densidad de Y:
	
	\begin{equation}
	f_2(y) = \displaystyle\int_{y}^{1-y} \frac{4}{289} \ dx = \frac{4}{289} \displaystyle\int_{y}^{1-y} 1 \ dx = \frac{4}{289}(1-2y) = \frac{4}{289}-\frac{8}{289}y, \ \ -8 < y < 0.5
	\end{equation}
	
	Calculo ahora las densidades condicionadas a ciertos valores $x_0, y_0$.
	
	\begin{equation}
	f_{Y/X=x_0}(y/x_0)= \frac{f(x_0,y)}{f_X(x_0)} = \left\lbrace
	\begin{array}{ll}
	\frac{\frac{4}{289}}{\frac{4x_0+32}{289}}= \frac{4}{4x_0+32} = \frac{1}{x_0+8}, \ \ -8 < x_0 < 0.5 \\ 
	\frac{\frac{4}{289}}{\frac{36-4x_0}{289}}= \frac{4}{36-4x_0} = \frac{1}{9-x_0}, \ \ 0.5 \leq x_0 < 9 \\ 
	\end{array}
	\ \ -8 < y < 0.5
	\right.
	\end{equation} 
	
	\begin{equation}
	f_{X/Y=y_0}(x/y_0)= \frac{f(x,y_0)}{f_Y(y_0)} = 
	\frac{\frac{4}{289}}{\frac{4-8y_0}{289}}= \frac{4}{4-8y_0} = \frac{1}{1-2y_0}, \ \ -8 < y_0 < 0.5, \ -8 < x < 9 \\ 
	\end{equation} 
	
	Finalmente, calculamos las probabilidades condicionadas deseadas:
	
	\begin{equation}
	P(Y > -3.38|X = 3.81) = \displaystyle\int_{-3.38}^{+\infty} f_{Y/X=3.81}(y) \ dy = \displaystyle\int_{-3.38}^{1-x} \frac{1}{9-3.81} dy = \displaystyle\int_{-3.38}^{-2.81} \frac{1}{5.19} dy =  \frac{1}{5.19}(-2.81+3.38) = 0.1098
	\end{equation}
	
	\begin{equation}
	P(X < 3.81|Y = -3.38) = \displaystyle\int_{-\infty}^{3.81} f_{X/Y=-3.38}(x) \ dx = \displaystyle\int_{y}^{3.81} \frac{1}{1-2*(-3.38)} dx = \displaystyle\int_{-3.38}^{3.81} \frac{1}{7.76} dx = \frac{1}{7.76}(3.81+3.38) = 0.927
	\end{equation}
	
\end{sol}
